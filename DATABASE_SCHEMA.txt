| handle_attendance_points  | CREATE OR REPLACE FUNCTION public.handle_attendance_points()
|                           | RETURNS trigger
|                           | LANGUAGE plpgsql
|                           | SECURITY DEFINER
|                           |AS $function$
|                           |BEGIN
|                           |  -- VALIDASI BARU: HANYA yang 'hadir' (Tepat Waktu) yang dapet poin
|                           |  IF (NEW.status = 'hadir') THEN
|                           |    
|                           |    -- 1. Update Poin di Profil
|                           |    UPDATE public.profiles
|                           |    SET points = points + 1
|                           |    WHERE id = NEW.profile_id;
|                           |    
|                           |    -- 2. Catat Riwayat Mutasi Poin
|                           |    INSERT INTO public.point_history (user_id, amount, type, description)
|                           |    VALUES (NEW.profile_id, 1, 'earning', 'Poin kehadiran: HADIR TEPAT WAKTU');
|                           |    
|                           |  END IF;
|                           |  
|                           |  RETURN NEW;
|                           |END;
|                           |$function$

| process_product_order     | CREATE OR REPLACE FUNCTION public.process_product_order()
|                           | RETURNS trigger
|                           | LANGUAGE plpgsql
|                           | SECURITY DEFINER
|                           |AS $function$
|                           |DECLARE
|                           |  product_price INTEGER;
|                           |  user_points INTEGER;
|                           |  product_stock INTEGER; -- Tambahan variabel stok
|                           |BEGIN
|                           |  -- 1. Ambil harga dan stok produk
|                           |  SELECT price, stock INTO product_price, product_stock 
|                           |  FROM public.products WHERE id = NEW.product_id;
|                           |  
|                           |  -- 2. Ambil poin user saat ini
|                           |  SELECT points INTO user_points FROM public.profiles WHERE id = NEW.user_id;
|                           |
|                           |  -- 3. VALIDASI 1: Poin cukup nggak?
|                           |  IF user_points < product_price THEN
|                           |    RAISE EXCEPTION 'Poin tidak cukup untuk membeli produk ini!';
|                           |  END IF;
|                           |
|                           |  -- 4. VALIDASI 2: Stok ada nggak?
|                           |  IF product_stock <= 0 THEN
|                           |    RAISE EXCEPTION 'Maaf, stok produk ini sudah habis!';
|                           |  END IF;
|                           |
|                           |  -- 5. POTONG POIN
|                           |  UPDATE public.profiles
|                           |  SET points = points - product_price
|                           |  WHERE id = NEW.user_id;
|                           |
|                           |  -- 6. POTONG STOK (Ini yang tadi lupa!)
|                           |  UPDATE public.products
|                           |  SET stock = stock - 1
|                           |  WHERE id = NEW.product_id;
|                           |
|                           |  -- 7. CATAT RIWAYAT
|                           |  INSERT INTO public.point_history (user_id, amount, type, description, order_id)
|                           |  VALUES (NEW.user_id, -product_price, 'spending', 'Penukaran hadiah ID: ' || NEW.product_id, NEW.id);
|                           |
|                           |  RETURN NEW;
|                           |END;
|                           |$function$
|                           |
| handle_order_cancellation | CREATE OR REPLACE FUNCTION public.handle_order_cancellation()
|                           | RETURNS trigger
|                           | LANGUAGE plpgsql
|                           | SECURITY DEFINER
|                           |AS $function$
|                           |BEGIN
|                           |  -- PERBAIKAN: Menggunakan 'dibatalkan' sesuai dengan Check Constraint di tabel product_orders
|                           |  IF NEW.status = 'dibatalkan' AND OLD.status != 'dibatalkan' THEN
|                           |    
|                           |    -- 1. Balikin Poin ke Profil Siswa
|                           |    UPDATE public.profiles 
|                           |    SET points = points + (SELECT price FROM public.products WHERE id = NEW.product_id)
|                           |    WHERE id = NEW.user_id;
|                           |
|                           |    -- 2. Balikin Stok Barang ke Toko
|                           |    UPDATE public.products 
|                           |    SET stock = stock + 1 
|                           |    WHERE id = NEW.product_id;
|                           |
|                           |    -- 3. Catat di History Mutasi Poin (Dioptimasi dengan order_id)
|                           |    INSERT INTO public.point_history (user_id, amount, type, description, order_id)
|                           |    VALUES (
|                           |      NEW.user_id, 
|                           |      (SELECT price FROM public.products WHERE id = NEW.product_id), 
|                           |      'earning', 
|                           |      'Refund: Pesanan dibatalkan (Order ID: ' || NEW.id || ')',
|                           |      NEW.id
|                           |    );
|                           |    
|                           |  END IF;
|                           |
|                           |  RETURN NEW;
|                           |END;
|                           |$function$
